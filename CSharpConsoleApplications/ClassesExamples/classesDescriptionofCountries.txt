    struct Temperature {
        public double? January;
        public double? July;
        public double? lowest;
        public Temperature(double january, double july, double lowest) {
            this.January = january;
            this.July = july;
            this.lowest = lowest;
        }
    }

public long Ukrainian;

public long Georgian;
